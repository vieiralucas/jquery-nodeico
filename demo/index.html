jQuery Nodeico

A jQuery plugin to display your Node Modules on NPM.

                
$('#example1').nodeico({
    name: 'express',
    downloads: true,
    rank: true,
    stars: true
});    
                
            
                
$('#example2').nodeico({
    name: 'express',
}); 
                
            
                
$('#example3').nodeico({
    name: 'express',
    type: 'compact'
}); 
                
            
                
$('#example4').nodeico({
    name: 'express',
    type: 'mini'
});
